-- Query per generare tabella furgoni:

CREATE TABLE furgoni(
    id integer PRIMARY KEY AUTOINCREMENT,
    price float NOT NULL,
    brand varchar(50) NOT NULL,
    model varchar(50) NOT NULL,
    km int,
    image_path varchar NOT NULL
);


-- Query per generare tabella auto:

CREATE TABLE auto(
    id integer PRIMARY KEY AUTOINCREMENT,
    price float NOT NULL,
    brand varchar(50) NOT NULL,
    model varchar(50) NOT NULL,
    km int,
    image_path varchar NOT NULL
);


-- Query per generare tabella moto

CREATE TABLE moto(
    id integer PRIMARY KEY AUTOINCREMENT,
    price float NOT NULL,
    brand varchar(50) NOT NULL,
    model varchar(50) NOT NULL,
    km int,
    cc int NOT NULL,
    image_path varchar NOT NULL
);


-- Query per generare tabella dei noleggi

CREATE TABLE rental(
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    furgone_id INTEGER,
    auto_id INTEGER,
    moto_id INTEGER,
    price INT NOT NULL,
    has_full_insurance BOOLEAN NOT NULL,
    has_navetta BOOLEAN DEFAULT 0,
    has_seggiolino BOOLEAN DEFAULT 0,
    has_secondo_conducente BOOLEAN DEFAULT 0,
    FOREIGN KEY (user_id) REFERENCES user(id),
    FOREIGN KEY (furgone_id) REFERENCES furgone(id),
    FOREIGN KEY (auto_id) REFERENCES auto(id),
    FOREIGN KEY (moto_id) REFERENCES moto(id)
);



-- Query per generare tabella degli utenti

CREATE TABLE user(
    id integer PRIMARY KEY AUTOINCREMENT,
    role VARCHAR(20) NOT NULL DEFAULT 'user',
    first_name varchar(50) NOT NULL,
    surname varchar(50) NOT NULL,
    email varchar(50) NOT NULL,
    user_password varchar(20) NOT NULL,
    prefix varchar(50) NOT NULL,
    phone_number varchar(50) NOT NULL,
    birth_year varchar(50) NOT NULL,
    birth_month varchar(50) NOT NULL,
    birth_day varchar(50) NOT NULL,
    driver_license_country varchar(50) NOT NULL,
    driver_license_number varchar(50) NOT NULL,
    driver_license_year varchar(50) NOT NULL,
    driver_license_month varchar(50) NOT NULL,
    driver_license_day varchar(50) NOT NULL,
    id_card_number varchar(50) NOT NULL,
    id_card_country varchar(50) NOT NULL,
    id_card_year varchar(50) NOT NULL,
    id_card_month varchar(50) NOT NULL,
    id_card_day varchar(50) NOT NULL,
    credit_card_number varchar(50) NOT NULL,
    credit_card_year varchar(50) NOT NULL,
    credit_card_month varchar(50) NOT NULL,
    billing_address varchar(50) NOT NULL,
    billing_address_city varchar(50) NOT NULL,
    billing_address_country varchar(50) NOT NULL,
    zip_code varchar(50) NOT NULL,
    email_consent boolean default 0
);

CREATE TABLE newsletter(
    id integer PRIMARY KEY AUTOINCREMENT,
    email varchar(50) NOT NULL
);

-- Query per generare tabella dei prefissi

CREATE TABLE prefixes(
    id integer PRIMARY KEY AUTOINCREMENT,
    prefix varchar(5) NOT NULL,
    country varchar(50) NOT NULL
);
